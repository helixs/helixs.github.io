下载并编译aria2

最新版本 

 1wget https://github.com/aria2/aria2/releases/download/release-1.34.0/aria2-1.34.0.tar.gz
 2tar -zxvf aria2-1.34.0.tar.gz 
 3cd aria2-1.34.0 
 4 #安装GNU编译器套件 
 5yum install gcc*
 6./configure 
 7make 
 8make install
 9cd src
10cd src cp aria2c /usr/bin

配置成系统服务

 1vim /etc/init.d/aria2c
 2----------
 3#!/bin/sh
 4### BEGIN INIT INFO
 5# Provides: aria2
 6# Required-Start: $remote_fs $network
 7# Required-Stop: $remote_fs $network
 8# Default-Start: 2 3 4 5
 9# Default-Stop: 0 1 6
10# Short-Description: Aria2 Downloader
11### END INIT INFO
12case "$1" in
13start)
14  echo -n "Starting aria2c"
15  /usr/local/bin/aria2c --conf-path=/etc/aria2/aria2.conf -D
16  ;;
17stop)
18  echo -n "Shutting down aria2c "
19  killall aria2c ;;
20restart)
21  killall aria2c
22  /usr/local/bin/aria2c --conf-path=/etc/aria2/aria2.conf -D
23  ;;
24esac
25exit
26

1chmod 755 /etc/init.d/aria2c
